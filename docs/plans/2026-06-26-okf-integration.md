# OKF Integration — Implementation Plan

> **For Hermes:** Use subagent-driven-development skill to implement this plan task-by-task.

**Goal:** Integrate Open Knowledge Format (OKF) from GoogleCloudPlatform/knowledge-catalog into rekipedia as three standalone features: OKF export format, wiki changelog (log.md), and per-section index.md for progressive disclosure.

**Architecture:**
- All three features are additive (no breaking changes to existing exports).
- OKF exporter follows the existing exporter pattern: a class in `src/rekipedia/exporters/okf_export.py` + wired into `src/rekipedia/cli/export.py`.
- `log.md` is written by `MarkdownExporter.export()` on every scan/update, appended by `reki update`.
- `index.md` per section is written alongside wiki pages during scan, generated from frontmatter `section` groupings.

**Tech Stack:** Python 3.11, YAML frontmatter (PyYAML already in deps), pathlib, click, pytest, rich

**Reference:** OKF SPEC — https://github.com/GoogleCloudPlatform/knowledge-catalog/blob/main/okf/SPEC.md

### Task 1.2: Create `OkfExporter` class with frontmatter mapper

**Objective:** Implement `OkfExporter` that converts rekipedia wiki pages to OKF Concept Documents.

**Files:**
- Create: `src/rekipedia/exporters/okf_export.py`

**OKF frontmatter schema** (from SPEC §4.1):
```yaml
type:        # REQUIRED — e.g. "rekipedia/wiki-page", "rekipedia/diagram"
title:       # human-readable name
description: # one-line summary (first non-empty line of body, stripped of #)
resource:    # relative path within the bundle
tags:        # list[str]
timestamp:   # ISO 8601 last-modified time
```

**Implementation:**
```python
# src/rekipedia/exporters/okf_export.py
"""OKF (Open Knowledge Format) exporter for rekipedia wikis.

Spec: https://github.com/GoogleCloudPlatform/knowledge-catalog/blob/main/okf/SPEC.md
"""
from __future__ import annotations

import re
from datetime import UTC, datetime
from pathlib import Path
from typing import Any

import yaml


BUNDLE_INDEX_HEADER = """\
# rekipedia Knowledge Bundle

This bundle was generated by [rekipedia](https://github.com/unrealandychan/rekipedia)
and conforms to the [Open Knowledge Format (OKF) v0.1](https://github.com/GoogleCloudPlatform/knowledge-catalog/blob/main/okf/SPEC.md).

"""


def _to_okf_frontmatter(meta: dict[str, Any]) -> dict[str, Any]:
    """Map rekipedia page metadata → OKF frontmatter fields."""
    slug = meta.get("slug", "")
    return {
        "type": "rekipedia/wiki-page",
        "title": meta.get("title", slug.replace("-", " ").title()),
        "description": meta.get("description", ""),
        "resource": f"wiki/{slug}.md",
        "tags": meta.get("tags", []),
        "timestamp": meta.get("created_at", datetime.now(UTC).isoformat()),
    }


def _extract_description(body: str) -> str:
    """Extract first meaningful non-heading line as description."""
    for line in body.splitlines():
        stripped = line.strip()
        if stripped and not stripped.startswith("#") and not stripped.startswith("---"):
            return stripped[:200]
    return ""


def _strip_reki_frontmatter(content: str) -> tuple[dict, str]:
    """Split rekipedia wiki page into (frontmatter_dict, body_str)."""
    if not content.startswith("---"):
        return {}, content
    end = content.find("\n---", 3)
    if end == -1:
        return {}, content
    fm_str = content[3:end].strip()
    body = content[end + 4:].lstrip("\n")
    try:
        fm = yaml.safe_load(fm_str) or {}
    except Exception:
        fm = {}
    return fm, body


def _write_okf_page(content: str, meta: dict, dest: Path) -> None:
    """Write a single OKF concept document with OKF frontmatter."""
    reki_fm, body = _strip_reki_frontmatter(content)
    # Merge: reki frontmatter overrides meta dict
    merged = {**meta, **reki_fm}
    okf_fm = _to_okf_frontmatter(merged)
    if not okf_fm["description"]:
        okf_fm["description"] = _extract_description(body)
    fm_str = yaml.dump(okf_fm, allow_unicode=True, sort_keys=False, default_flow_style=False)
    dest.write_text(f"---\n{fm_str}---\n\n{body}", encoding="utf-8")


class OkfExporter:
    """Export rekipedia wiki as an OKF Knowledge Bundle directory."""

    def __init__(self, wiki_dir: Path, diagrams_dir: Path, repo_root: Path) -> None:
        self.wiki_dir = Path(wiki_dir)
        self.diagrams_dir = Path(diagrams_dir)
        self.repo_root = Path(repo_root)

    def export(self, output_dir: Path, pages_meta: dict[str, dict] | None = None) -> dict:
        """Write OKF bundle to output_dir. Returns summary dict."""
        output_dir = Path(output_dir)
        wiki_out = output_dir / "wiki"
        diagrams_out = output_dir / "diagrams"
        wiki_out.mkdir(parents=True, exist_ok=True)
        diagrams_out.mkdir(parents=True, exist_ok=True)
        pages_meta = pages_meta or {}

        written_pages: list[str] = []
        written_diagrams: list[str] = []

        # Export wiki pages
        if self.wiki_dir.exists():
            for md_file in sorted(self.wiki_dir.glob("*.md")):
                slug = md_file.stem
                content = md_file.read_text(encoding="utf-8")
                meta = pages_meta.get(slug, {"slug": slug})
                meta["slug"] = slug
                dest = wiki_out / md_file.name
                _write_okf_page(content, meta, dest)
                written_pages.append(slug)

        # Export diagrams (type: rekipedia/diagram)
        if self.diagrams_dir.exists():
            for md_file in sorted(self.diagrams_dir.glob("*.md")):
                slug = md_file.stem
                content = md_file.read_text(encoding="utf-8")
                fm = {
                    "type": "rekipedia/diagram",
                    "title": slug.replace("-", " ").title(),
                    "description": f"Mermaid diagram: {slug}",
                    "resource": f"diagrams/{slug}.md",
                    "tags": ["diagram"],
                    "timestamp": datetime.now(UTC).isoformat(),
                }
                fm_str = yaml.dump(fm, allow_unicode=True, sort_keys=False, default_flow_style=False)
                dest = diagrams_out / md_file.name
                dest.write_text(f"---\n{fm_str}---\n\n{content}", encoding="utf-8")
                written_diagrams.append(slug)

        # Write bundle index.md (OKF §6)
        index_lines = [BUNDLE_INDEX_HEADER, "## Wiki Pages\n\n"]
        for slug in written_pages:
            meta = pages_meta.get(slug, {})
            title = meta.get("title", slug.replace("-", " ").title())
            index_lines.append(f"- [{title}](wiki/{slug}.md)\n")
        if written_diagrams:
            index_lines.append("\n## Diagrams\n\n")
            for slug in written_diagrams:
                index_lines.append(f"- [{slug.replace('-', ' ').title()}](diagrams/{slug}.md)\n")
        (output_dir / "index.md").write_text("".join(index_lines), encoding="utf-8")

        # Write log.md (OKF §7) — initial entry
        now = datetime.now(UTC).isoformat()
        log_entry = f"## {now}\n\nOKF bundle exported by rekipedia from `{self.repo_root.name}`.\n\n- {len(written_pages)} wiki pages\n- {len(written_diagrams)} diagrams\n\n"
        log_path = output_dir / "log.md"
        if log_path.exists():
            existing = log_path.read_text(encoding="utf-8")
            log_path.write_text(log_entry + existing, encoding="utf-8")
        else:
            log_path.write_text(f"# Knowledge Bundle Changelog\n\n{log_entry}", encoding="utf-8")

        return {
            "pages": written_pages,
            "diagrams": written_diagrams,
            "bundle_dir": str(output_dir),
        }
```

def _write_log_entry(wiki_dir: Path, run_id: str | None, pages: dict) -> None:
    """Append a changelog entry to wiki/log.md (OKF §7 pattern)."""
    log_path = wiki_dir / "log.md"
    now = datetime.now(UTC).strftime("%Y-%m-%dT%H:%M:%SZ")
    run_label = run_id or "unknown"
    new_pages = sorted(pages.keys())
    page_list = "\n".join(f"  - `{slug}`" for slug in new_pages)
    entry = f"## {now} (run: `{run_label}`)\n\n- Pages updated: {len(new_pages)}\n{page_list}\n\n"

    if log_path.exists():
        existing = log_path.read_text(encoding="utf-8")
        # Strip header if present, prepend new entry
        if existing.startswith("# Wiki Changelog"):
            header, rest = existing.split("\n", 1)
            log_path.write_text(f"{header}\n\n{entry}{rest.lstrip()}", encoding="utf-8")
        else:
            log_path.write_text(entry + existing, encoding="utf-8")
    else:
        log_path.write_text(f"# Wiki Changelog\n\n{entry}", encoding="utf-8")

### Task 3.2: Create `section_index.py`

**File:** `src/rekipedia/exporters/section_index.py`

```python
"""Generate per-section index.md files for progressive disclosure (OKF §6 pattern)."""
from __future__ import annotations

from collections import defaultdict
from pathlib import Path
from typing import Any


_SECTION_INDEX_TEMPLATE = """\
---
type: rekipedia/section-index
title: "{section_title}"
description: "Index of all {count} pages in the '{section}' section."
tags: [index, {section}]
---

# {section_title}

This section index lists all wiki pages in the **{section}** category.

## Pages

{page_list}

---
*Generated by [rekipedia](https://github.com/unrealandychan/rekipedia). Do not edit manually.*
"""


def write_section_indexes(
    wiki_dir: Path,
    pages_meta: list[dict[str, Any]],
) -> dict[str, Any]:
    """Write one `index-<section>.md` per section into wiki_dir.

    Args:
        wiki_dir: Path to the .rekipedia/wiki/ directory.
        pages_meta: List of page metadata dicts (from manifest.json pages list).

    Returns:
        dict with key "sections": {section_name: page_count}
    """
    wiki_dir = Path(wiki_dir)
    sections: dict[str, list[dict]] = defaultdict(list)

    for meta in pages_meta:
        section = (meta.get("section") or "").strip()
        if not section:
            continue
        sections[section].append(meta)

    written: dict[str, int] = {}
    for section, pages in sections.items():
        section_title = section.replace("-", " ").replace("_", " ").title()
        page_lines = []
        for page in sorted(pages, key=lambda p: p.get("title", p["slug"])):
            slug = page["slug"]
            title = page.get("title", slug.replace("-", " ").title())
            tags = page.get("tags", [])
            tag_str = f" — `{'`, `'.join(tags)}`" if tags else ""
            page_lines.append(f"- [{title}]({slug}.md){tag_str}")

        page_list = "\n".join(page_lines)
        content = _SECTION_INDEX_TEMPLATE.format(
            section=section,
            section_title=section_title,
            count=len(pages),
            page_list=page_list,
        )
        dest = wiki_dir / f"index-{section}.md"
        dest.write_text(content, encoding="utf-8")
        written[section] = len(pages)

    return {"sections": written}
```
