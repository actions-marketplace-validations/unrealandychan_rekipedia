```python
"""Pure-Python Louvain-inspired community detection for rekipedia.

No external graph library required — uses a label-propagation
approach that converges in O(V+E) per pass for sparse graphs.
"""
from __future__ import annotations

from collections import defaultdict


def detect_communities(edges: list[tuple[str, str]]) -> dict[str, int]:
    """Assign community IDs to nodes via label propagation.

    Args:
        edges: List of (from_node, to_node) string tuples.

    Returns:
        Dict mapping node_name → community_id (int, 0-indexed).
        Nodes that appear only as self-loops get their own community.
        Empty input returns empty dict.
    """
    if not edges:
        return {}

    # Build adjacency list (treat directed as undirected for community)
    neighbours: dict[str, set[str]] = defaultdict(set)
    all_nodes: set[str] = set()
    for frm, to in edges:
        all_nodes.add(frm)
        all_nodes.add(to)
        if frm != to:
            neighbours[frm].add(to)
            neighbours[to].add(frm)

    # Initialise: each node is its own community
    labels: dict[str, str] = {n: n for n in all_nodes}

    # Label propagation — up to 20 passes or until stable
    for _ in range(20):
        changed = False
        for node in sorted(all_nodes):  # sorted → deterministic
            nbrs = neighbours[node]
            if not nbrs:
                continue
            # Pick the most common label among neighbours
            freq: dict[str, int] = defaultdict(int)
            for nb in nbrs:
                freq[labels[nb]] += 1
            best = max(freq, key=lambda k: (freq[k], k))  # tie-break: lexicographic
            if labels[node] != best:
                labels[node] = best
                changed = True
        if not changed:
            break

    # Convert string labels → stable integer IDs
    unique_labels = sorted(set(labels.values()))
    label_to_id = {lbl: i for i, lbl in enumerate(unique_labels)}
    return {node: label_to_id[labels[node]] for node in all_nodes}
```

```python
class CommunityShardPlanner(ShardPlanner):
    """Shard planner that respects import-graph community structure.

    Files in the same community are co-located in the same shard
    before the token-budget split kicks in. This keeps semantically
    related files together so the LLM sees complete context for each
    cluster of modules.
    """

    def plan_with_communities(
        self,
        files: list[FileManifest],
        community_map: dict[str, int],
        llm: LLMConfig | None = None,
    ) -> list[Shard]:
        """Plan shards respecting community membership.

        Args:
            files: All file manifests to shard.
            community_map: Mapping of file path → community id.
                           Files absent from the map are treated as
                           their own community (id = hash(path)).
            llm: Optional LLM config (forwarded to Shard objects).

        Returns:
            List of Shard objects with community-aware grouping.
        """
        if not files:
            return []

        llm = llm or LLMConfig()

        # Group files by community id
        community_groups: dict[int, list[FileManifest]] = {}
        for f in files:
            cid = community_map.get(f.path, hash(f.path))
            community_groups.setdefault(cid, []).append(f)

        # For each community group, apply token-budget splitting
        shards: list[Shard] = []
        for cid in sorted(community_groups.keys()):
            group = community_groups[cid]
            group_dir = f"community_{cid}"
            shards.extend(self._split_group(group_dir, group, llm))

        return shards
```

```python
def compute_shortest_path(
    from_symbol: str,
    to_symbol: str,
    relationships: list,
) -> dict:
    """BFS shortest path between two symbols in the relationship graph.

    Args:
        from_symbol: Source symbol name.
        to_symbol: Target symbol name.
        relationships: List of relationship dicts or Relationship objects.

    Returns:
        Dict with keys:
        - ``path``: list[str] from source to target, or None if unreachable.
        - ``length``: int hop count, or None if unreachable.
    """
    from collections import deque

    # Build adjacency: directed graph
    graph: dict[str, list[str]] = {}
    for rel in relationships:
        if isinstance(rel, dict):
            frm  = rel.get("from_", "") or rel.get("from", "")
            to   = rel.get("to", "")
        else:
            frm  = rel.from_ or ""
            to   = rel.to or ""
        if frm and to:
            graph.setdefault(frm, []).append(to)

    if from_symbol not in graph:
        return {"path": None, "length": None}

    # BFS
    queue: deque[list[str]] = deque([[from_symbol]])
    visited: set[str] = {from_symbol}
    while queue:
        path = queue.popleft()
        current = path[-1]
        if current == to_symbol:
            return {"path": path, "length": len(path) - 1}
        for neighbour in graph.get(current, []):
            if neighbour not in visited:
                visited.add(neighbour)
                queue.append(path + [neighbour])

    return {"path": None, "length": None}
```
